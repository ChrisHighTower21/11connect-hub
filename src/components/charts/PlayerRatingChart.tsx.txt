"use client";

import {
    Chart as ChartJS,
    CategoryScale,
    LinearScale,
    PointElement,
    LineElement,
    Tooltip,
    Legend,
} from "chart.js";

import { Line } from "react-chartjs-2";

ChartJS.register(
    CategoryScale,
    LinearScale,
    PointElement,
    LineElement,
    Tooltip,
    Legend
);

type Match = {
    date: string;
    rating: number;
};

type Props = {
    matches: Match[];
};

export function PlayerRatingChart({ matches }: Props) {
    return (
        <Line
            data={{
                labels: matches.map((m) => m.date),
                datasets: [
                    {
                        label: "Bewertung",
                        data: matches.map((m) => m.rating),
                        tension: 0.35,
                    },
                ],
            }}
            options={{
                responsive: true,
                plugins: {
                    legend: {
                        display: false,
                    },
                },
                scales: {
                    y: {
                        min: 0,
                        max: 10,
                    },
                },
            }}
        />
    );
}
